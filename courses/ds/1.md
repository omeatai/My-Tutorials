# O(n)

```js
const nemo = ['nemo'];

const large = new Array(5).fill('nemo');

function findNemo(array) {
  let count = 0;
  for (let i = 0; i < array.length; i++) {
    if (array[i] === 'nemo') {
      console.log('Found NEMO!');
    }
    count += 1;
  }
  console.log(`O(${count})`)
}

findNemo(large); // O(n) ----> Linear Time
```
